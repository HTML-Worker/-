    "no-ex-assign":"error",

//不允许重新分配function的声明

    "no-func-assign":"error",

    示例：
    /*-- bad --*/
      function foo() {}
        foo = bar;

      function foo() {
        foo = bar;
      }

      /*-- good --*/
        var foo = function () {}
        foo = bar;

        function foo(foo) { // `foo` is shadowed.
          foo = bar;
        }

        function foo() {
          var foo = bar;  // `foo` is shadowed.
        }

    /*-- bad --*/
    let foo = function() {

          return 1;
      } catch(err) {
          return 2;
      } finally {
          return 3;
      }
    };

    /*-- good --*/
    let foo = function() {
      try {
          return 1;
      } catch(err) {
          return 2;
      } finally {
          let a = function() {
              return "hola!";
          }
      }
    };
